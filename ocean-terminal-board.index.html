<script>

// ===========================

// FULL 24-HOUR BOARD GENERATOR

// ===========================

const tramTimesTable = document.getElementById('tramTimes');

// Weekday rules: array of objects {start, end, stepMinutes}

const weekdayIntervals = [

  { start: "05:23", end: "07:03", step: 10 },

  { start: "07:03", end: "20:09", step: 7 },

  { start: "20:09", end: "23:59", step: 10 }

];

// Weekend rules

const weekendIntervals = [

  { start: "05:23", end: "09:03", step: 10 },

  { start: "09:03", end: "22:09", step: 7 },

  { start: "22:09", end: "23:59", step: 10 }

];

// Helper to convert HH:MM → minutes

function timeToMinutes(t) {

  const [h,m] = t.split(":").map(Number);

  return h*60 + m;

}

// Helper to convert minutes → HH:MM

function minutesToTime(m) {

  const h = Math.floor(m/60).toString().padStart(2,'0');

  const min = (m%60).toString().padStart(2,'0');

  return `${h}:${min}`;

}

// Generate tr rows for an interval

function generateIntervalRows(intervals) {

  const today = new Date();

  const day = today.getDay(); // 0 = Sunday, 1=Monday, ..., 6=Saturday

  let rules = (day === 0 || day === 6) ? weekendIntervals : weekdayIntervals;

  rules.forEach(r => {

    let startM = timeToMinutes(r.start);

    const endM = timeToMinutes(r.end);

    while(startM <= endM) {

      const t = minutesToTime(startM);

      const tr = document.createElement('tr');

      tr.dataset.time = t;

      tr.innerHTML = `<td>Ocean Terminal</td><td>${t}</td>`;

      tramTimesTable.appendChild(tr);

      startM += r.step;

    }

  });

}

// Highlight next tram

function updateNextTram() {

  const now = new Date();

  const hours = now.getHours().toString().padStart(2,'0');

  const minutes = now.getMinutes().toString().padStart(2,'0');

  const currentTime = `${hours}:${minutes}`;

 

  const rows = document.querySelectorAll('#tramTimes tr[data-time]');

  rows.forEach(row => row.classList.remove('next'));

  for (let row of rows) {

    if (row.dataset.time >= currentTime) {

      row.classList.add('next');

      break; // only highlight next tram

    }

  }

}

// Run everything

generateIntervalRows();

updateNextTram();

setInterval(updateNextTram, 30000);
